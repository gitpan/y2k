Overloading: Bizarre Y2K Detection Example

        package y2k;
        ...
        sub to_num {
          my ($year) = @_;
          return $year->{YEAR};
        }

        sub concat {
          my ($y2k, $s, $rev) = @_;
          carp("Detected possible Y2K problem");
          my $year = $y2k->{YEAR} % 100         # Or omit %100.
          $rev ? $s . $year : $year . $s;
        }
